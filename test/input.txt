function int fibonacci(int n) {
    if (n <= 1) {
        return n;
    } else {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

function void print_fibonacci_table(int limit) {

    for (i = 1; i <= limit; i++) {
        for (j = 1; j <= limit; j = j*2){
            print(fibonacci(i * j));

    }
}
